javascript// Horizontal Scroll for Hero â†’ Slogan transition
let horizontalScrollHandler = null;

function setupHorizontalScroll() {
    const outer = document.querySelector('.horizontal-scroll-outer');
    const wrapper = document.querySelector('.horizontal-scroll-wrapper');
    const container = document.querySelector('.horizontal-scroll-container');
    
    if (!outer || !wrapper || !container) return;
    
    // 2ê°œ ì„¹ì…˜ (íˆì–´ë¡œ ë³µì œ + ìŠ¬ë¡œê±´)
    const totalSections = 2;
    const totalHeight = totalSections * window.innerHeight;
    
    outer.style.height = totalHeight + 'px';
    
    function handleScroll() {
        const outerRect = outer.getBoundingClientRect();
        const scrollStart = -outerRect.top;
        const scrollEnd = totalHeight - window.innerHeight;
        
        if (scrollStart >= 0 && scrollStart <= scrollEnd) {
            const progress = scrollStart / scrollEnd;
            const translateX = -progress * window.innerWidth;
            
            container.style.transform = `translateX(${translateX}px)`;
        }
    }
    
    if (horizontalScrollHandler) {
        window.removeEventListener('scroll', horizontalScrollHandler);
        if (window.lenis) {
            window.lenis.off('scroll', horizontalScrollHandler);
        }
    }
    
    horizontalScrollHandler = handleScroll;
    window.addEventListener('scroll', handleScroll, { passive: true });
    
    if (window.lenis) {
        window.lenis.on('scroll', handleScroll);
    }
    
    handleScroll();
}

let equipmentScrollHandler = null;

function setupEquipmentNarrative() {
    const narrative = document.querySelector('.equipment-narrative');
    const stickyImage = document.getElementById('equipment-image');
    const steps = document.querySelectorAll('.equipment-step');
    
    if (!narrative || !stickyImage || steps.length === 0) return;
    
    const equipmentData = siteData.content.diagnosis.details;
    
    function handleScroll() {
        const narrativeRect = narrative.getBoundingClientRect();
        const triggerPoint = window.innerHeight / 2;
        
        steps.forEach((step, index) => {
            const stepRect = step.getBoundingClientRect();
            const stepCenter = stepRect.top + stepRect.height / 2;
            
            if (Math.abs(stepCenter - triggerPoint) < window.innerHeight / 3) {
                // í™œì„±í™”
                steps.forEach(s => s.classList.remove('active'));
                step.classList.add('active');
                
                // ì´ë¯¸ì§€ ì „í™˜
                const currentImg = stickyImage.querySelector('img');
                const newSrc = equipmentData[index].img;
                
                if (currentImg.src.indexOf(newSrc) === -1) {
                    currentImg.style.opacity = '0';
                    setTimeout(() => {
                        currentImg.src = newSrc;
                        currentImg.style.opacity = '1';
                    }, 300);
                }
            }
        });
    }
    
    if (equipmentScrollHandler) {
        window.removeEventListener('scroll', equipmentScrollHandler);
        if (window.lenis) {
            window.lenis.off('scroll', equipmentScrollHandler);
        }

}
